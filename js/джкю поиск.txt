$(document).ready(function search() {
    $('#b_search').click(function (event) {
      event.preventDefault(); 
        var subStr = $('#substr').val().toLowerCase();//содержимое пол€ ввода
        $('h3').each(function () {
            var aTag = $(this),// слово в текущее колеции 
                i = (aTag.attr('counter') !== undefined) ? aTag.attr('counter') : 0,// проверка на наличее счЄтчика
                innerText = aTag.text().toLowerCase();//выт€гиваем текст
            aTag.removeClass('findenow');//убераем класс
            if (innerText.indexOf(subStr) != -1) {//провер€ем чтобы элемент сушействовал
                aTag.addClass('findenow').attr('counter', ++i);// добавл€ем класс и счЄтчик
              }
            if (i >= 3) {//ожидаем 3 совпадени€ и болеее
                aTag.addClass('most-finde');// добавл€ем класс
              
            }
        });
    });
});
